# PackMule Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a rule-based auto-assignment system (PackMule) to KRT: when the master looter opens a loot window, rules match items (by itemID or quality threshold) and a preview popup lets the raid leader confirm auto-distribution in one click.

**Architecture:** New `Services/Loot/PackMule/` module with three files — `Rules.lua` (persistence + validation, no side-effects), `Matcher.lua` (pure match function), `Service.lua` (orchestrator wired via the existing event Bus). Preview is a new controller + XML popup. Rule editor is a new tab in the existing `Widgets/Config.lua`. Reserves and active rolls always win: items in either state are skipped before the matcher runs.

**Tech Stack:** Lua 5.1, WoW 3.3.5a client API, existing KRT infrastructure (`addon.Core.GetFeatureShared()`, `Bus` event pub/sub, SavedVariables, `Diag` logging).

**Spec:** [docs/superpowers/specs/2026-04-21-packmule-design.md](../specs/2026-04-21-packmule-design.md)

```lua
-- ----- KRT Lua Contract ----- --
-- deps: local addon = select(2, ...)
-- shared: persistence + validation for PackMule rules
-- exports: addon.Services.Loot.PackMule._Rules
-- events: none

local addon = select(2, ...)
local feature = addon.Core.GetFeatureShared()
local PM = feature.C.PACKMULE

addon.Services = addon.Services or {}
addon.Services.Loot = addon.Services.Loot or {}
addon.Services.Loot.PackMule = addon.Services.Loot.PackMule or {}
local module = addon.Services.Loot.PackMule
module._Rules = module._Rules or {}
local Rules = module._Rules

local tonumber, tostring = tonumber, tostring
local type = type

local function genId()
    return string.format("%d-%d", math.floor((type(time) == "function" and time() or os.time())), math.random(100000, 999999))
end

local function isValidAction(action)
    if type(action) ~= "table" then return false end
    if action.type == PM.ACTION_SKIP then return true end
    if action.type == PM.ACTION_ASSIGN then
        return type(action.target) == "string" and action.target ~= ""
    end
    return false
end

local function normalizeItemRule(r)
    if type(r) ~= "table" then return nil end
    local itemId = tonumber(r.itemId)
    if not itemId or itemId <= 0 then return nil end
    if not isValidAction(r.action) then return nil end
    return {
        id = tostring(r.id or genId()),
        kind = PM.KIND_ITEM,
        enabled = r.enabled ~= false,
        itemId = itemId,
        itemName = tostring(r.itemName or ""),
        action = { type = r.action.type, target = r.action.target },
        createdAt = tonumber(r.createdAt) or 0,
    }
end

local function normalizeQualityRule(r)
    if type(r) ~= "table" then return nil end
    local q = tonumber(r.qualityMax)
    if not q or q < PM.QUALITY_MIN or q > PM.QUALITY_MAX then return nil end
    if not isValidAction(r.action) then return nil end
    return {
        id = tostring(r.id or genId()),
        kind = PM.KIND_QUALITY,
        enabled = r.enabled ~= false,
        qualityMax = q,
        action = { type = r.action.type, target = r.action.target },
        createdAt = tonumber(r.createdAt) or 0,
    }
end

function Rules.Normalize(store)
    if type(store) ~= "table" then
        return { version = PM.SAVEDVAR_VERSION, enabled = true, itemRules = {}, qualityRules = {} }
    end
    local out = {
        version = PM.SAVEDVAR_VERSION,
        enabled = store.enabled ~= false,
        itemRules = {},
        qualityRules = {},
    }
    if type(store.itemRules) == "table" then
        for _, r in ipairs(store.itemRules) do
            local n = normalizeItemRule(r)
            if n then out.itemRules[#out.itemRules + 1] = n end
        end
    end
    if type(store.qualityRules) == "table" then
        for _, r in ipairs(store.qualityRules) do
            local n = normalizeQualityRule(r)
            if n then out.qualityRules[#out.qualityRules + 1] = n end
        end
    end
    return out
end

function Rules.AddItemRule(store, data)
    local r = normalizeItemRule({
        id = genId(),
        itemId = data.itemId,
        itemName = data.itemName,
        action = data.action,
        createdAt = type(time) == "function" and time() or os.time(),
    })
    if not r then return nil end
    store.itemRules[#store.itemRules + 1] = r
    return r.id
end

function Rules.AddQualityRule(store, data)
    local r = normalizeQualityRule({
        id = genId(),
        qualityMax = data.qualityMax,
        action = data.action,
        createdAt = type(time) == "function" and time() or os.time(),
    })
    if not r then return nil end
    store.qualityRules[#store.qualityRules + 1] = r
    return r.id
end

function Rules.RemoveRule(store, id)
    for i, r in ipairs(store.itemRules) do
        if r.id == id then table.remove(store.itemRules, i); return true end
    end
    for i, r in ipairs(store.qualityRules) do
        if r.id == id then table.remove(store.qualityRules, i); return true end
    end
    return false
end

function Rules.ToggleRule(store, id, enabled)
    for _, list in ipairs({ store.itemRules, store.qualityRules }) do
        for _, r in ipairs(list) do
            if r.id == id then r.enabled = enabled and true or false; return true end
        end
    end
    return false
end
```

```lua
-- ItemID beats Quality (O2)

    qualityRules = { { enabled = true, qualityMax = 5, action = { type = "assign", target = "Bob" } } },
```

```lua
-- ----- KRT Lua Contract ----- --
-- deps: local addon = select(2, ...)
-- shared: pure matching engine: (item, rules) → action | nil
-- exports: addon.Services.Loot.PackMule._Matcher
-- events: none

local addon = select(2, ...)
local feature = addon.Core.GetFeatureShared()

addon.Services = addon.Services or {}
addon.Services.Loot = addon.Services.Loot or {}
addon.Services.Loot.PackMule = addon.Services.Loot.PackMule or {}
local module = addon.Services.Loot.PackMule
module._Matcher = module._Matcher or {}
local Matcher = module._Matcher

local type = type
local tonumber = tonumber

function Matcher.Match(item, rules)
    if type(item) ~= "table" or type(rules) ~= "table" then return nil end
    if rules.enabled == false then return nil end
    if item.locked == true then return nil end

    local itemId = tonumber(item.itemId)
    local quality = tonumber(item.quality)

    if itemId and type(rules.itemRules) == "table" then
        for i = 1, #rules.itemRules do
            local r = rules.itemRules[i]
            if r and r.enabled ~= false and tonumber(r.itemId) == itemId then
                return { type = r.action.type, target = r.action.target }
            end
        end
    end

    if quality and type(rules.qualityRules) == "table" then
        for i = 1, #rules.qualityRules do
            local r = rules.qualityRules[i]
            if r and r.enabled ~= false and quality <= tonumber(r.qualityMax) then
                return { type = r.action.type, target = r.action.target }
            end
        end
    end

    return nil
end
```

```lua
-- ----- KRT Lua Contract ----- --
-- deps: local addon = select(2, ...)
-- shared: PackMule orchestrator: subscribe LootFetched → Evaluate → emit PackMuleMatched
-- exports: addon.Services.Loot.PackMule
-- events: in  LootFetched; out PackMuleMatched, PackMuleExecuted, PackMuleCancelled

local addon = select(2, ...)
local feature = addon.Core.GetFeatureShared()

local Diag = feature.Diag
local C = feature.C
local Bus = feature.Bus or addon.Bus
local Events = feature.Events or addon.Events
local Services = feature.Services or addon.Services

addon.Services.Loot.PackMule = addon.Services.Loot.PackMule or {}
local module = addon.Services.Loot.PackMule
local Rules = assert(module._Rules, "PackMule.Rules not initialized")
local Matcher = assert(module._Matcher, "PackMule.Matcher not initialized")

local PM = C.PACKMULE
local InternalEvents = Events.Internal

local tonumber, tostring, type = tonumber, tostring, type

-- Pending matches awaiting user confirmation.
local pending = { matches = {}, active = false }

local function getStore()
    KRT_PackMule = Rules.Normalize(KRT_PackMule)
    return KRT_PackMule
end

local function getDisenchanter()
    return addon.options and addon.options.disenchanter or nil
end

local function resolveTarget(target)
    if target == PM.DISENCHANTER_SENTINEL then
        return getDisenchanter()
    end
    return target
end

local function isReserved(itemId, itemLink)
    local R = Services.Reserves
    if not R or not R.GetReserveCountForItem then return false end
    return (R:GetReserveCountForItem(itemId, itemLink) or 0) > 0
end

local function hasActiveRoll(itemId)
    local R = Services.Rolls
    if not R or not R.HasActiveSessionFor then return false end
    return R:HasActiveSessionFor(itemId) == true
end

local function isIgnored(itemId)
    local I = addon.IgnoredItems
    return I and I.Ids and I.Ids[tonumber(itemId) or -1] == true
end

function module:EvaluateLoot()
    pending.matches = {}
    pending.active = false

    local Raid = Services.Raid
    if not Raid or not Raid.IsMasterLooter or not Raid:IsMasterLooter() then return end

    local Loot = Services.Loot
    if not Loot or not Loot.GetAllItems then return end
    local items = Loot:GetAllItems() or {}

    local store = getStore()
    if store.enabled == false then return end

    for _, item in ipairs(items) do
        local itemId = tonumber(item.itemId)
        if itemId and not isReserved(itemId, item.itemLink)
            and not hasActiveRoll(itemId)
            and not isIgnored(itemId) then
            local action = Matcher.Match(item, store)
            if action then
                local resolvedTarget = action.type == PM.ACTION_ASSIGN and resolveTarget(action.target) or nil
                local unresolved = action.type == PM.ACTION_ASSIGN and (not resolvedTarget or resolvedTarget == "")
                pending.matches[#pending.matches + 1] = {
                    slot = item.slot,
                    itemId = itemId,
                    itemLink = item.itemLink,
                    action = { type = action.type, target = action.target },
                    resolvedTarget = resolvedTarget,
                    unresolved = unresolved or false,
                    selected = not unresolved,
                }
            end
        end
    end

    if #pending.matches > 0 then
        pending.active = true
        Bus:Publish(InternalEvents.PackMuleMatched, { matches = pending.matches })
        if addon.hasDebug then
            addon:debug(Diag.D.LogPackMuleMatched:format(#pending.matches))
        end
    end
end

function module:GetPendingMatches()
    return pending.active and pending.matches or {}
end

function module:SetMatchSelected(index, selected)
    local m = pending.matches[index]
    if m and not m.unresolved then m.selected = selected and true or false end
end

function module:ExecuteMatches()
    if not pending.active then return end
    local Loot = Services.Loot
    local results = {}
    for _, m in ipairs(pending.matches) do
        if m.selected then
            if m.action.type == PM.ACTION_SKIP then
                results[#results + 1] = { slot = m.slot, itemLink = m.itemLink, target = nil, status = "skipped" }
            else
                local target = m.resolvedTarget
                local idx = target and Loot and Loot.GetCandidateIndex and Loot:GetCandidateIndex(m.slot, target) or nil
                if idx then
                    GiveMasterLoot(m.slot, idx)
                    results[#results + 1] = { slot = m.slot, itemLink = m.itemLink, target = target, status = "ok" }
                else
                    results[#results + 1] = { slot = m.slot, itemLink = m.itemLink, target = target, status = "ineligible" }
                end
            end
        end
    end
    Bus:Publish(InternalEvents.PackMuleExecuted, { results = results })
    if addon.hasDebug then
        addon:debug(Diag.D.LogPackMuleExecuted:format(#results))
    end
    pending.matches = {}
    pending.active = false
end

function module:CancelMatches()
    pending.matches = {}
    pending.active = false
    Bus:Publish(InternalEvents.PackMuleCancelled, {})
    if addon.hasDebug then
        addon:debug(Diag.D.LogPackMuleCancelled:format(""))
    end
end

-- Subscribe to loot fetch events.
if Bus and Bus.Subscribe and InternalEvents and InternalEvents.LootFetched then
    Bus:Subscribe(InternalEvents.LootFetched, function(payload)
        if payload and payload.source == "LOOT_OPENED" then
            module:EvaluateLoot()
        end
    end)
end

-- Cancel pending matches when the loot window closes.
if Events and Events.Wow and Events.Wow.LootClosed and Bus and Bus.Subscribe then
    Bus:Subscribe(Events.Wow.LootClosed, function()
        if pending.active then module:CancelMatches() end
    end)
end
```

- [ ] **Step 1: Create the controller**

```lua
-- ----- KRT Lua Contract ----- --
-- deps: local addon = select(2, ...)
-- shared: controller for PackMule preview popup
-- exports: addon.Controllers.PackMuleConfirm
-- events: in PackMuleMatched, PackMuleCancelled; out (calls) PackMule:ExecuteMatches/CancelMatches

local addon = select(2, ...)
local feature = addon.Core.GetFeatureShared()
local L = feature.L
local C = feature.C
local Bus = feature.Bus or addon.Bus
local Events = feature.Events or addon.Events
local Services = feature.Services or addon.Services

addon.Controllers = addon.Controllers or {}
local module = {}
addon.Controllers.PackMuleConfirm = module

local PM = C.PACKMULE
local InternalEvents = Events.Internal

local frame                      -- KRTPackMuleConfirm XML frame
local rowPool = {}               -- reusable row frames
local timeoutHandle              -- C_Timer style cancel

local function ensureFrame()
    if frame then return frame end
    frame = _G.KRTPackMuleConfirm
    if not frame then return nil end
    frame:SetScript("OnHide", function() module:OnHide() end)
    _G.KRTPackMuleConfirmButton:SetText(L.BtnPackMuleConfirm)
    _G.KRTPackMuleConfirmButton:SetScript("OnClick", function() module:OnConfirm() end)
    _G.KRTPackMuleConfirmCancelButton:SetText(L.BtnPackMuleCancel)
    _G.KRTPackMuleConfirmCancelButton:SetScript("OnClick", function() module:OnCancel() end)
    return frame
end

local function getRow(index)
    if rowPool[index] then return rowPool[index] end
    local parent = _G.KRTPackMuleConfirmScrollContent
    if not parent then return nil end
    local row = CreateFrame("Frame", nil, parent)
    row:SetHeight(22)
    row:SetWidth(parent:GetWidth() - 10)
    row.check = CreateFrame("CheckButton", nil, row, "UICheckButtonTemplate")
    row.check:SetPoint("LEFT", 0, 0)
    row.label = row:CreateFontString(nil, "OVERLAY", "GameFontNormal")
    row.label:SetPoint("LEFT", row.check, "RIGHT", 4, 0)
    row.check:SetScript("OnClick", function(btn)
        Services.Loot.PackMule:SetMatchSelected(index, btn:GetChecked() and true or false)
        module:UpdateConfirmLabel()
    end)
    rowPool[index] = row
    return row
end

local function hideUnusedRows(startIdx)
    for i = startIdx, #rowPool do rowPool[i]:Hide() end
end

function module:UpdateConfirmLabel()
    local btn = _G.KRTPackMuleConfirmButton
    if not btn then return end
    local matches = Services.Loot.PackMule:GetPendingMatches()
    local n = 0
    for _, m in ipairs(matches) do if m.selected then n = n + 1 end end
    btn:SetText(string.format("%s (%d)", L.BtnPackMuleConfirm, n))
    if n == 0 then btn:Disable() else btn:Enable() end
end

function module:Render(matches)
    if not ensureFrame() then return end
    _G.KRTPackMuleConfirmTitle:SetText(L.StrPackMuleTitle:format(#matches))

    local yOffset = -4
    for i, m in ipairs(matches) do
        local row = getRow(i)
        if row then
            row:SetPoint("TOPLEFT", _G.KRTPackMuleConfirmScrollContent, "TOPLEFT", 4, yOffset)
            row.check:SetChecked(m.selected)
            if m.unresolved then
                row.check:Disable()
                row.label:SetText(string.format("|cffff4040%s → %s|r", m.itemLink or "?", L.StrPackMuleUnresolved))
            else
                row.check:Enable()
                local targetLabel = m.action.type == PM.ACTION_SKIP and L.StrPackMuleSkip
                    or (m.action.target == PM.DISENCHANTER_SENTINEL
                        and string.format("%s (DE)", m.resolvedTarget or "?")
                        or m.resolvedTarget or "?")
                row.label:SetText(string.format("%s → %s", m.itemLink or "?", targetLabel))
            end
            row:Show()
            yOffset = yOffset - 24
        end
    end
    hideUnusedRows(#matches + 1)
    self:UpdateConfirmLabel()
    frame:Show()
    self:ArmTimeout()
end

function module:ArmTimeout()
    self:CancelTimeout()
    if type(C_Timer) == "table" and type(C_Timer.NewTimer) == "function" then
        timeoutHandle = C_Timer.NewTimer(PM.PREVIEW_TIMEOUT_SECONDS, function() module:OnCancel() end)
    else
        -- 3.3.5a fallback: use addon:ScheduleTimer if available
        if addon.ScheduleTimer then
            timeoutHandle = addon:ScheduleTimer(function() module:OnCancel() end, PM.PREVIEW_TIMEOUT_SECONDS)
        end
    end
end

function module:CancelTimeout()
    if not timeoutHandle then return end
    if type(timeoutHandle) == "table" and timeoutHandle.Cancel then
        timeoutHandle:Cancel()
    elseif addon.CancelTimer then
        addon:CancelTimer(timeoutHandle)
    end
    timeoutHandle = nil
end

function module:OnConfirm()
    self:CancelTimeout()
    Services.Loot.PackMule:ExecuteMatches()
    if frame then frame:Hide() end
end

function module:OnCancel()
    self:CancelTimeout()
    Services.Loot.PackMule:CancelMatches()
    if frame then frame:Hide() end
end

function module:OnHide()
    self:CancelTimeout()
end

-- Event wiring
if Bus and Bus.Subscribe then
    Bus:Subscribe(InternalEvents.PackMuleMatched, function(payload)
        module:Render(payload.matches or {})
    end)
    Bus:Subscribe(InternalEvents.PackMuleCancelled, function()
        if frame then frame:Hide() end
    end)
end
```
